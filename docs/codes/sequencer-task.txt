loop {
    if let Ok(item) = sequencer_channel.recv() {
        if sequencer.enabled() {
            match item {
                SequencerMessage::NoteOn(step) => {
                    sequencer.step_trigger_on(step);
                }
                SequencerMessage::NoteOff(step) => {
                    sequencer.step_trigger_off(step);
                }
            }
        }
    }
}
